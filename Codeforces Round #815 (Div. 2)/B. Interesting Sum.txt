#include<iostream>
#include<bits/stdc++.h>
#define ll long long int

using namespace std;

int main(){
    int t;
    cin >> t;
    for (int i = 0; i < t; i++){
        int n;
        cin >> n;
        ll arr[n];
        for (int j = 0; j < n; j++){
            cin >> arr[j];
        }

        sort(arr, arr+n);

        ll ans;
        ans = arr[n-1] - arr[0] + arr[n-2] - arr[1];
        cout << ans << endl;
    }
}
