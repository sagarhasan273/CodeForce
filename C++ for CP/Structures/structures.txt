#include<iostream>
using namespace std;

struct car{
    int year;
    string model;
    string brand;
};

int main(){
    car myCar1;
    myCar1.brand = "BMW";
    myCar1.model = "s1000";
    myCar1.year = 1995;

    car myCar2;
    myCar2.brand = "Ford";
    myCar2.model = "Mustang";
    myCar2.year = 1969;

    cout << myCar1.brand << " " << myCar1.model << " " << myCar1.year << endl;
    cout << myCar2.brand << " " << myCar2.model << " " << myCar2.year << endl;
}
